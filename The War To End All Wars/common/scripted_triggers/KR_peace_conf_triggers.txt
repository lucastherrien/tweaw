restrict_territorial_ambitions = {
	OR = {
		# is a puppet
		is_subject = yes
		# is a faction member and leader is a major and at the conference
		AND = {
			is_in_faction = yes
			NOT = { is_faction_leader = yes }
			any_allied_country = {
				is_faction_leader = yes
				is_actual_major = yes
				at_conference_no_war_with_ROOT = yes
			}
		}
		# is not a major and a major is at the conference
		AND = {
			NOT = { is_actual_major = yes }
			any_other_country = {
				is_actual_major = yes
				at_conference_no_war_with_ROOT = yes
			}
		}
		# is not a major and no major is at the conference, but a much stronger minor is
		AND = {
			NOT = {
				is_actual_major = yes
				any_other_country = {
					is_actual_major = yes
					at_conference_no_war_with_ROOT = yes
				}
			}
			any_other_country = {
				at_conference_no_war_with_ROOT = yes
				strength_ratio = { tag = ROOT ratio > 1.5 }
				NOT = {
					is_actual_major = yes
					PREV_is_restriction_exception = yes
					is_subject = yes
				}
			}
		}
		# is an exception
		ROOT_is_restriction_exception = yes
	}
}

# whether a country will refuse to take states they otherwise can (country scope)
is_willing_to_take_state = {
	NOT = {
		# is a major who won't take the core/claim of an ally
		AND = {
			is_actual_major = yes
			FROM = {
				is_not_claim_or_core_of_PREV = yes
				NOT = { is_important_state = yes }
			}
			any_country = {
				NOT = {
					tag = PREV
					has_war_with = PREV
				}
				is_in_peace_conference = yes
				OR = {
					is_subject_of = PREV
					is_in_faction_with = PREV
				}
				FROM = { is_claim_or_core_of_PREV = yes }
			}
		}
		# is a major who won't take the core/claim of a friend
		AND = {
			is_actual_major = yes
			FROM = {
				is_not_claim_or_core_of_PREV = yes
				NOT = { is_important_state = yes }
			}
			NOT = {
				ambition_area = yes
				AND = {
					tag = GER
					FROM = { is_on_continent = europe }
				}
			}
			any_country = {
				is_in_peace_conference = yes
				FROM = { is_claim_or_core_of_PREV = yes }
				PREV = { has_opinion = { target = PREV value > 50 } }
				NOT = {
					tag = PREV
					has_war_with = PREV
					is_subject_of = PREV
					is_in_faction_with = PREV
				}
			}
		}
		# is a minor who won't take cores/claims of others
		AND = {
			NOT = { tag = MON }
			NOT = { is_actual_major = yes }
			FROM = { is_not_claim_or_core_of_PREV = yes }
			any_country = {
				NOT = {
					tag = PREV
					has_war_with = PREV
				}
				is_in_peace_conference = yes
				FROM = { is_claim_or_core_of_PREV = yes }
			}
		}
		# won't take an ally's ambition area
		AND = {
			FROM = { is_not_claim_or_core_of_PREV = yes }
			any_allied_country = {
				is_in_peace_conference = yes
				is_in_faction_with = PREV
				FROM = { is_not_claim_or_core_of_PREV = yes }
				ambition_area = yes
			}
			NOT = {
				ambition_area = yes
				AND = {
					OR = {
						tag = GER
						tag = NFA
					}
					FROM = {
						OR = {
							region = 19 #Northern France
							region = 20 #Southern France
						}
					}
				}
				AND = {
					tag = CAN
					FROM = {
						OR = {
							region = 33 #Alaska
							state = 685 #Panama Canal
							region = 186 #Northeast
							region = 120 #Midwest
							region = 117 #Southeast
							region = 119 #Southwest
							region = 118 #West Coast
							region = 185 #Northwest
						}
					}
					ROOT = {
						OR = {
							tag = NEE
							tag = CAL
						}
					}
					OR = {
						TEX = { at_conference_no_war_with_PREV = yes }
						CSA = { at_conference_no_war_with_PREV = yes }
						CAL = { at_conference_no_war_with_PREV = yes }
					}
				}
			}
		}
		# is a minor giving way to an allied major's ambition area
		AND = {
			NOT = { is_actual_major = yes }
			FROM = { is_not_claim_or_core_of_PREV = yes }
			any_allied_country = {
				is_in_peace_conference = yes
				is_actual_major = yes
				is_in_faction_with = PREV
				FROM = { is_not_claim_or_core_of_PREV = yes }
				ambition_area = yes
			}
			ambition_area = yes
		}
		# is unwilling to take land on another continent
		AND = {
			NOT = {
				same_continent_as_state_FROM = yes
				FROM = { is_claim_or_core_of_PREV = yes }
				ambition_area = yes
				is_exception_to_taking_states_on_other_continents = yes
			}
			any_country = {
				NOT = { tag = PREV }
				at_conference_has_war_with_owner = yes
				same_continent_as_state_FROM = yes
				OR = {
					FROM = { is_claim_or_core_of_PREV = yes }
					ambition_area = yes
					NOT = { restrict_territorial_ambitions_PREV = yes }
				}
			}
		}
		# is in faction with Canada and it's the UK
		AND = {
			NOT = { tag = CAN }
			is_in_faction_with = CAN
			FROM = {
				is_not_claim_or_core_of_PREV = yes
				OR = {
					region = 1 #Southern England
					region = 2 #Northern England
					region = 3 #Scotland
				}
			}
		}
	}
}

# countries that are willing to take states on other continents (country scope)
is_exception_to_taking_states_on_other_continents = {
	OR = {
		# European countries will always take states in Africa
		AND = {
			is_in_europe = yes
			NOT = { tag = RUS }
			NOT = { tag = SOV }
			FROM = { is_on_continent = africa }
		}
		# European countries will always take states in Asia and the Middle East
		# so long as there is no major there in their faction
		AND = {
			is_in_europe = yes
			NOT = { tag = RUS }
			NOT = { tag = SOV }
			FROM = { is_on_continent = asia }
			NOT = {
				any_allied_country = {
					is_actual_major = yes
					is_in_peace_conference = yes
					is_in_asia = yes
				}
			}
		}
		AND = {
			is_in_europe = yes
			FROM = { is_on_continent = middle_east }
			NOT = {
				any_allied_country = {
					is_actual_major = yes
					is_in_peace_conference = yes
					is_in_middle_east = yes
				}
			}
		}
		# Russia will take non-island states in Asia and the Middle East, and islands if it has an eastern port
		AND = {
			OR = {
				tag = RUS
				tag = SOV
			}
			OR = {
				any_owned_state = { region = 148 } #Amur Region
				FROM = {
					NOT = {
						is_island_state = yes
						region = 154 #Home Islands
						region = 160 #Phillipines
						is_core_of = INS
					}
				}
			}
			FROM = { is_on_continent = asia }
		}
		# Middle Eastern countries will take states in northern India and North Africa
		AND = {
			is_in_middle_east = yes
			FROM = {
				OR = {
					state = 441 #Kashmir
					state = 444 #Baluchistan
					state = 443 #Sindh
					state = 445 #Quetta
					state = 442 #Peshawar
					state = 891 #Lahore
					region = 126 #North Africa
					region = 189 #Maghreb
					region = 128 #Sudan
					region = 17 #Horn of Africa
					state = 183 #Cyprus
				}
			}
		}
		# countries in North Africa will take states in the Middle East
		AND = {
			NOT = { tag = NFA }
			capital_scope = {
				OR = {
					region = 126 #North Africa
					region = 189 #Maghreb
				}
			}
			FROM = {
				OR = {
					is_on_continent = middle_east
					state = 116 #Malta
					state = 182 #Crete
					state = 183 #Cyprus
				}
			}
		}
		# countries in India will take states in the middle east
		AND = {
			capital_scope = {
				OR = {
					region = 153 #Northern India
					region = 141 #Eastern India
					region = 31 #Southern India
				}
			}
			FROM = { is_on_continent = middle_east }
		}
		# Turkey/Ottomans will take states in the Balkans
		AND = {
			OR = {
				tag = OTT
				tag = TUR
			}
			FROM = {
				OR = {
					region = 69 #Eastern Mediterranean
					region = 24 #Western Balkans
					region = 25 #Greece
					region = 26 #Eastern Balkans
				}
			}
		}
		# Canada will always take states in Europe, N & S America if it is a faction leader
		AND = {
			tag = CAN
			is_faction_leader = yes
			FROM = {
				OR = {
					is_on_continent = north_america
					is_on_continent = south_america
					is_on_continent = europe
				}
			}
		}
		# National France will always take cores in Europe
		AND = {
			tag = NFA
			FROM = { is_on_continent = europe }
		}
		# American countries will always take states in S America
		AND = {
			OR = {
				tag = USA
				tag = CSA
				tag = TEX
				tag = CAL
			}
			FROM = { is_on_continent = south_america }
		}
		# Japan will take states in oceania
		AND = {
			tag = JAP
			FROM = { is_on_continent = australia }
		}
		# Germany will take states anywhere
		tag = GER
	}
}
