					ideology = syndicalist

					name = CSA_syndicalist_syndicalism_party
				}
				set_party_name = {
					ideology = radical_socialist
					long_name = CSA_radical_socialist_party_long
					name = CSA_radical_socialist_party
				}
			}
		}
		hidden_effect = {
			if = {
				limit = { has_country_flag = CSA_socialist }
				set_party_name = {
					ideology = totalist
					long_name = CSA_totalist_federalist_party_long
					name = CSA_totalist_federalist_party
				}
				set_party_name = {
					ideology = syndicalist

					ideology = radical_socialist
					long_name = CSA_radical_socialist_federalist_party_long
					name = CSA_radical_socialist_federalist_party
				}
			}
		}
		if = { limit = { has_country_flag = CSA_capital_chicago } set_capital = 395 }
		if = { limit = { has_country_flag = CSA_capital_washington } set_capital = 361 }
		if = { limit = { has_country_flag = CSA_capital_philly } set_capital = 360 }

	}
