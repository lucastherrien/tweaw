add_namespace = asiapeace

country_event = { #Reichspakt is defeated in Asia
	id = asiapeace.1
	immediate = {log = "[GetDateText]: [Root.GetName]: event asiapeace.1"}
	title = asiapeace.1.t
	desc = asiapeace.1.d

	is_triggered_only = yes
	picture = placeholder

	trigger = {
		tag = GER
		has_war = yes
		is_faction_leader = yes
		any_enemy_country = {
			is_in_asia = yes
			has_capitulated = no
			OR = {
				is_faction_leader = yes
				is_in_faction = no
			}
		}
		any_allied_country = {
			is_in_asia = yes
		}
		NOT = {
			any_allied_country = {
				is_in_asia = yes
				has_capitulated = no
			}
		}
	}

	option = { #We surrender!
		name = asiapeace.1.a
		ai_chance = { factor = 100 }
		custom_effect_tooltip = asian_reichspakt_will_proceed_to_peacedeal_we_white_peace
		hidden_effect = {
			every_country = {
				limit = {
					is_in_faction_with = GER
					is_in_asia = yes
					has_capitulated = yes
				}
				if = {
					limit = { is_subject = yes }
					overlord = {
						end_puppet = PREV
					}
				}
				leave_faction = yes
			}
		}
		every_country = {
			limit = {
				is_in_asia = yes
				OR = {
					is_in_faction = no
					is_faction_leader = yes
				}
				has_war_with = GER
				is_subject = no
			}
			set_country_flag = asian_adversary_for_germany
			country_event = { id = asiapeace.2 }
		}
	}

	option = { #We shall keep fighting
		name = asiapeace.1.b
		ai_chance = { factor = 0 }
		add_war_support = -0.1
		add_stability = -0.1
	}
}
