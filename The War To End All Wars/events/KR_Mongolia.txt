	title = mongolia.11.t
	desc = mongolia.11.d
	picture = GFX_report_event_spain_civil_war_soldiers

	fire_only_once = yes

	trigger = {
		tag = MON
		owns_state = 618
		NOT = { country_exists = XSM }
	}

	option = {
		name = mongolia.11.a

		ai_chance = { factor = 95 }

		add_manpower = -5000

		add_resource = { type = steel amount = 10 state = 330 }

		add_state_core = 616 #Inner Mongolia
		add_state_core = 617 #Urumqi
		add_state_core = 618 #Dzungaria
		add_state_core = 1083 #Kumul
		add_state_core = 774 #North Gansu
		add_state_core = 348 #Ningxia

		every_owned_state = {
			limit = {
				is_core_of = XSM
				NOT = { is_core_of = MON }
			}
			add_claim_by = MON
		}

		if = {
			limit = {
				NOT = { is_in_faction_with = TIB }
			}

			every_state = {
				limit = {
					is_core_of = TIB
					NOT = { is_claimed_by = MON }
					NOT = { is_core_of = MON }
				}
				add_claim_by = MON
			}
		}

		if = {
			limit = {
				is_in_faction_with = TIB
			}

			every_state = {
				limit = { is_owned_by = TIB }

				remove_claim_by = MON
			}
		}

		set_global_flag = MON_take_XSM
	}

	option = {
		name = mongolia.11.b

		ai_chance = { factor = 5 }

		add_stability = -0.10

		add_state_core = 616
		add_state_core = 617
		add_state_core = 618

			add_claim_by = MON
		}

		if = {
			limit = {
				NOT = { is_in_faction_with = TIB }
			}

			every_state = {
				limit = { is_core_of = TIB }

				add_claim_by = MON
			}
		}

				is_in_faction_with = TIB
