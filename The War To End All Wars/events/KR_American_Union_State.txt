country_event = {# Socialist terrorism - factory sabotaged
	id = unionstate.200
	immediate = {log = "[GetDateText]: [Root.GetName]: event unionstate.200"}
	title = unionstate.200.t
	desc = unionstate.200.d
	picture = GFX_report_event_SRI_bombing

	trigger = {
		tag = TEX
		NOT = { has_idea = USA_socialist_attack }
		or = {
			has_idea = USA_socialist_resistance1
			has_idea = USA_socialist_resistance2
			has_idea = USA_socialist_resistance3
			has_idea = USA_socialist_resistance4
		}
		any_owned_state = {
			is_core_of = TEX
			industrial_complex > 0
			is_controlled_by = TEX
		}
	}

	mean_time_to_happen = { months = 3 }

	immediate = {
		hidden_effect = {
			random_owned_state = {
				limit = {
					OR = {
						is_core_of = USA
						is_core_of = CSA
						is_core_of = TEX
						is_core_of = CAL
					}
					industrial_complex > 0
					is_controlled_by = TEX
				}
				save_event_target_as = attack_state
			}
		}
	}

	option = {
		name = unionstate.200.a
		add_timed_idea = { idea = USA_socialist_attack days = 90 }
		add_stability = -0.01
		event_target:attack_state = {
			damage_building = {
				type = industrial_complex
				damage = 1.0
			}
		}
	}
}
