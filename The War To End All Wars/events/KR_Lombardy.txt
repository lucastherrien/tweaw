add_namespace = lombardy


### Ask for the Iron Crown of Lombardy
country_event = {
	id = lombardy.1
	immediate = {log = "[GetDateText]: [Root.GetName]: event lombardy.1"}
	title = lombardy.1.t
	desc = lombardy.1.d

	is_triggered_only = yes

	option = {
		name = lombardy.1.a
		trigger = {
			country_exists = AUS
		}
		AUS = {
			country_event = {
				id = lombardy.2
			}
		}
	}
	option = {
		name = lombardy.1.b
		trigger = {
			NOT = {
				country_exists = AUS
			}
			country_exists = SIC
		}
		SIC = {
			country_event = {
				id = lombardy.2
			}
		}
	}
	option = {
		name = lombardy.1.c
		trigger = {
			NOT = {
				OR = {
					country_exists = AUS
					country_exists = SIC
				}
			}
			ROOT = {
				is_subject = yes
			}
		}
		overlord = {
			country_event = {
				id = lombardy.2
			}
		}
	}
	option = {
		name = lombardy.1.e
		trigger = {
			NOT = {
				OR = {
					country_exists = AUS
					country_exists = SIC
					ROOT = {
						is_subject = yes
					}
				}
			}
			ROOT = {
				is_in_faction = yes
			}
		}
		random_other_country = {
			limit = {
				is_faction_leader = yes
				is_in_faction_with = LOM
			}
			country_event = {
				id = lombardy.2
			}
		}
	}
}

### Lombardy asks for the Iron Crown
country_event = {
	id = lombardy.2
	immediate = {log = "[GetDateText]: [Root.GetName]: event lombardy.2"}
	title = lombardy.2.t
	desc = lombardy.2.d

	is_triggered_only = yes

	#Return it
	option = {
		add_political_power = -10
		name = lombardy.2.a
		FROM = {
			country_event = {
				id = lombardy.3
				days = 7
			}
		}
		effect_tooltip = {
			LOM = {
				add_opinion_modifier = {
					target = ROOT
					modifier = LOM_returned_crown
				}
			}
		}
	}

	#Refuse!
	option = {
		name = lombardy.2.a
		FROM = {
			country_event = {
				id = lombardy.4
				days = 7
			}
		}
		effect_tooltip = {
			LOM = {
				add_opinion_modifier = {
					target = ROOT
					modifier = LOM_refused_crown
				}
			}
		}
	}
}

### Hire Military Advisors
country_event = {
	id = lombardy.6
	immediate = {log = "[GetDateText]: [Root.GetName]: event lombardy.6"}
	title = lombardy.6.t
	desc = lombardy.6.d

	is_triggered_only = yes

	option = { # Prussian advisors are the best
		trigger = {
			country_exists = GER
			NOT = {
				has_war_with = GER
			}
		}
		ai_chance = {
			factor = 50
			modifier = {
				if = {
					limit = {
						country_exists = GER
						OR = {
							is_in_faction_with = GER
							is_subject_of = GER
						}
					}
				}
				factor = 10
			}
		}
		name = lombardy.6.a
		add_ideas = LOM_Prussian_drill
		if = {
			limit = {
				country_exists = GER
			}
			add_opinion_modifier = {
				target = GER
				modifier = LOM_prussian_instructors
			}
			hidden_effect = {
				GER = {
					country_event = {
						id = lombardy.7
					}
				}
			}
		}
	}
	option = { # Italian advisors are popular
		ai_chance = {
			factor = 50
		}
		name = lombardy.6.b
		add_political_power = 10
		add_ideas = LOM_Italian_drill
	}
}
