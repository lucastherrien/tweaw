country_event = { #the AI is dumb, let's help it a bit
	id = russia.0
	immediate = {log = "[GetDateText]: [Root.GetName]: event russia.0"}
	#picture = GFX_petain_NFA

	fire_only_once = no

	hidden = yes

	trigger = {
		tag = RUS
		is_ai = yes
		date > 1937.3.1
		has_stability < 0.1
	}

	mean_time_to_happen = {
		days = 500
	}

	option = {
		if = {
			limit = {
				has_stability < -0.5
			}
			add_stability = 0.3
		}
		if = {
			limit = {
				has_stability < -0.3 has_stability > -0.5
			}
			add_stability = 0.262
		}
		if = {
			limit = {
				has_stability < -0.1 has_stability > -0.3
			}
			add_stability = 0.225
		}
		if = {
			limit = {
				has_stability < 0.1 has_stability > -0.1
			}
			add_stability = 0.15
		}
	}

}

country_event = { #change civil war ideas
	id = russia.2005
	immediate = {log = "[GetDateText]: [Root.GetName]: event russia.2005"}

	hidden = yes

	trigger = {
		tag = RUS
		has_war_with = SOV
		OR = {
			AND = {
				has_idea = rus_cw
				has_stability > 0
			}
			AND = {
				has_idea = rus_cw_very_low
				OR = {
					has_stability > 0.25
					has_stability < 0
				}
			}
			AND = {
				has_idea = rus_cw_low
				OR = {
					has_stability > 0.4
					has_stability < 0.25
				}
			}
			AND = {
				has_idea = rus_cw_average
				has_stability < 0.4
			}
		}
	}

	mean_time_to_happen = { days = 10 }

	immediate = {
		if = {
			limit = {
				NOT = { has_idea = rus_cw }
				has_stability < 0
			}
			remove_ideas = rus_cw_very_low
			remove_ideas = rus_cw_low
			remove_ideas = rus_cw_average
			add_ideas = rus_cw
		}
		if = {
			limit = {
				NOT = { has_idea = rus_cw_very_low }
				has_stability < 0.25
				has_stability > 0
			}
			remove_ideas = rus_cw
			remove_ideas = rus_cw_low
			remove_ideas = rus_cw_average
			add_ideas = rus_cw_very_low
		}
		if = {
			limit = {
				NOT = { has_idea = rus_cw_low }
				has_stability < 0.4
				has_stability > 0.25
			}
			remove_ideas = rus_cw_very_low
			remove_ideas = rus_cw
			remove_ideas = rus_cw_average
			add_ideas = rus_cw_low
		}
		if = {
			limit = {
				NOT = { has_idea = rus_cw_average }
				has_stability > 0.4
			}
			remove_ideas = rus_cw_very_low
			remove_ideas = rus_cw_low
			remove_ideas = rus_cw
			add_ideas = rus_cw_average
		}
	}
