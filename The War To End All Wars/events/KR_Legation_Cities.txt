			}

		set_country_flag = triads_seize_LEC_profits
	}

	# Nothing we can do
	option = {
		name = legcities.60.a
		ai_chance = {
			factor = 200
		}
		if = {
			limit = { has_idea = PROFF_LEG_CITIES }
			remove_ideas = PROFF_LEG_CITIES
			set_country_flag = had_full_proff_leg_cities
		}
		if = {
			limit = { has_idea = MINOR_PROFF_LEG_CITIES }
			remove_ideas = MINOR_PROFF_LEG_CITIES
			set_country_flag = had_minor_proff_leg_cities
		}
		if = {
			limit = { country_exists = LEC }
			if = {
				limit = { has_opinion_modifier = LEC_member }
				remove_opinion_modifier = { target = LEC modifier = LEC_member }
			}
			add_opinion_modifier = { target = LEC modifier = INVEST_SEIZED }
			if = {
				limit = { LEC = { is_guaranteed_by = ROOT } }
				diplomatic_relation = { country = LEC relation = guarantee active = no }
			}
			if = {
				limit = { LEC = { gives_military_access_to = ROOT } }
				diplomatic_relation = { country = LEC relation = military_access active = no }
			}
		}
	}

	#Time for an intervention
	option = {
		name = legcities.60.b
		ai_chance = {
			factor = 100
			modifier = {

				NOT = { tag = JAP }
			}
		}
		if = {
			limit = {
				has_idea = PROFF_LEG_CITIES
			}
			remove_ideas = PROFF_LEG_CITIES

			set_country_flag = LEC_foreign_intervention_planned
		}
		trigger = {
			OR = {
				AND = {
					is_major = yes
					OR = {
						has_war = no
						NOT = {
							any_enemy_country = {
								is_major = yes
								has_war_with = ROOT
							}
						}
					}
				}
				tag = JAP
			}
		}

		create_wargoal = {
			type = annex_everything
			target = LEC
		}
		add_political_power = -50
		if = {
			limit = {
				country_exists = LEC
			}
			if = {
				limit = {
					has_opinion_modifier = LEC_member
				}
				remove_opinion_modifier = {
					target = LEC modifier = LEC_member
				}
			}
			add_opinion_modifier = {
				target = LEC
				modifier = INVEST_SEIZED
			}
			diplomatic_relation = {
				country = LEC
				relation = guarantee
				active = no
			}
			diplomatic_relation = {
				country = LEC
				relation = military_access
				active = no
			}
			LEC = {
				country_event = {
					id = legcities.61
				}
			}
		}
	}
