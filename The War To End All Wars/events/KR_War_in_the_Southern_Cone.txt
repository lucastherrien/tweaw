country_event = {	#Argentina puppets Chile. Peru takes its claims
	id = southern.9
	immediate = {log = "[GetDateText]: [Root.GetName]: event southern.9"}
	title = southern.9.t
	desc = southern.9.d

	fire_only_once = yes

	trigger = {
		tag = ARG
		is_subject = no
		has_war_with = CHL
		OR = {
			is_in_faction = no
			is_faction_leader = yes
		}
		CHL = {
			surrender_progress > 0.8
			OR = {
				is_in_faction = no
				is_faction_leader = yes
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		hidden_effect = {
			CHL = {
				save_event_target_as = white_peace_receiver
				remove_ideas = ARG_CHILE_STOP_BEING_A_TWAT
			}
			remove_ideas = ARG_CHILE_STOP_BEING_A_TWAT
		}
	}

	option = {
		ai_chance = {
			factor = 100
		}
		name = southern.9.a
		#Release FOP to prevent shenanigans
		if = {
			limit = {
				FOP = { is_subject_of = CHL }
			}
			CHL = {
				end_puppet = FOP
			}
			FOP = {
				country_event = southern.91
			}
		}

		#peace out Chile
		ARG = { white_peace_with_TARGET = yes }
		PRU = {
			if = {
				limit = {
					exists = yes
					has_war_with = CHL
					OR = {
						is_in_faction = no
						is_faction_leader = yes
					}
				}
				white_peace_with_TARGET = yes
			}
		}
		BOL = {
			if = {
				limit = {
					exists = yes
					has_war_with = CHL
					OR = {
						is_in_faction = no
						is_faction_leader = yes
					}
				}
				white_peace_with_TARGET = yes
			}
		}

		#puppet remnants
		CHL = { destroy_ships = { type = destroyer count = 4 } }
		CHL = { destroy_ships = { type = light_cruiser count = 2 } }
		CHL = {
			transfer_navy = {
				target = ARG
			}
		}
		ARG = {
			annex_country = { target = CHL }
			release_puppet = CHL
		}

		#let Bolivia take land
		BOL = {
			if = {
				limit = {
					exists = yes
					has_global_flag = BOL_ARG
				}
				if = {
					limit = {
						CHL = {
							owns_state = 872
						}
					}
					transfer_state = 872
				}
				if = {
					limit = {
						CHL = {
							owns_state = 873
						}
					}
					transfer_state = 873
				}
				if = {
					limit = {
						CHL = {
							owns_state = 506
						}
					}
					transfer_state = 506
				}
			}
		}
	}

	option = {    #Annex Chile
		ai_chance = {
			factor = 0
		}
		name = southern.9.b

		add_stability = -0.1
		add_war_support = -0.1
		add_political_power = -150

		#Release FOP to prevent shenanigans
		if = {
			limit = {
				FOP = { is_subject_of = CHL }
			}
			CHL = {
				end_puppet = FOP
			}
			FOP = {
				country_event = southern.91
			}
		}

		#peace out Chile
		ARG = { white_peace_with_TARGET = yes }
		PRU = {
			if = {
				limit = {
					exists = yes
					has_war_with = CHL
					OR = {
						is_in_faction = no
						is_faction_leader = yes
					}
				}
				white_peace_with_TARGET = yes
			}
		}
		BOL = {
			if = {
				limit = {
					exists = yes
					has_war_with = CHL
					OR = {
						is_in_faction = no
						is_faction_leader = yes
					}
				}
				white_peace_with_TARGET = yes
			}
		}

		#puppet remnants
		CHL = { destroy_ships = { type = destroyer count = 4 } }
		CHL = { destroy_ships = { type = light_cruiser count = 2 } }
		CHL = {
			transfer_navy = {
				target = ARG
			}
		}

		#let Bolivia take land
		BOL = {
			if = {
				limit = {
					exists = yes
					has_global_flag = BOL_ARG
				}
				if = {
					limit = {
						CHL = {
							owns_state = 872
						}
					}
					transfer_state = 872
				}
				if = {
					limit = {
						CHL = {
							owns_state = 873
						}
					}
					transfer_state = 873
				}
				if = {
					limit = {
						CHL = {
							owns_state = 506
						}
					}
					transfer_state = 506
				}
			}
		}

		ARG = {
			annex_country = { target = CHL }
		}
	}
}

country_event = {	#Argentina takes Paraguay. Bolivia takes its claims
	id = southern.10
	immediate = {log = "[GetDateText]: [Root.GetName]: event southern.10"}
	title = southern.10.t
	desc = southern.10.d

	fire_only_once = yes

	trigger = {
		tag = ARG
		has_war_with = PAR
		is_subject = no
		surrender_progress < 0.30
		OR = {
			is_in_faction = no
			is_faction_leader = yes
		}
		PAR = {
			owns_state = 688 #Chaco Boreal
			surrender_progress > 0.90
			is_subject = no
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		#peace out Paraguay
		if = {
			limit = {
				688 = {
					is_owned_by = PAR
					is_controlled_by = BOL
				}
			}
			BOL = {
				transfer_state = 688
			}
		}
		PAR = { save_event_target_as = white_peace_receiver }
		ARG = { white_peace_with_TARGET = yes }
		BOL = {
			if = {
				limit = {
					exists = yes
					has_war_with = PAR
					OR = {
						is_in_faction = no
						is_faction_leader = yes
					}
				}
				white_peace_with_TARGET = yes
			}
		}

		#takes Paraguay and Chacos
		if = {
			limit = {
				1027 = { is_owned_by = PAR }
			}
			if = {
				limit = { country_exists = BOL }
				BOL = { transfer_state = 1027 }
				else = {
					ARG = { transfer_state = 1027 }
				}
			}
		}
		ARG = {
			transfer_state = 688
			if = {
				limit = {
					PAR = { owns_state = 301 }
				}
				transfer_state = 301
			}
		}
	}

	option = {	#Give them Chaco Boreal
		trigger = {
			BOL = {
				exists = yes
			}
		}
		ai_chance = {
			factor = 100
			modifier = {
				factor = 3
				has_global_flag = BOL_ARG
			}
		}
		name = southern.10.a
		BOL = {
			transfer_state = 688
		}
	}

	option = {	#take Chaco Boreal
		ai_chance = {
			factor = 100
		}
		name = southern.10.b
		if = {
			limit = {
				BOL = {
					exists = yes
					is_subject = no
				}
			}
			BOL = {
				country_event = southern.11
			}
		}
	}
}
