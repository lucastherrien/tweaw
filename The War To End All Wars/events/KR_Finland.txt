	id = finland.1
	immediate = {log = "[GetDateText]: [Root.GetName]: event finland.1"}

	id = finland.2
	immediate = {log = "[GetDateText]: [Root.GetName]: event finland.2"}
	title = finland.2.t
	desc = finland.2.d

	fire_only_once = yes

	trigger = {
		tag = FIN
		is_subject = no
		owns_state = 110
		NOT = {
			country_exists = NOR
		}
	}

	mean_time_to_happen = {
		days = 7
	}

	option = {
		name = finland.2.a #Liberate Norway
		ai_chance = {
			factor = 20
		}
		every_state = {
			limit = {
				is_core_of = NOR
				is_controlled_by = ROOT
