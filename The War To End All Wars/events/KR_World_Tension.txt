# Second Weltkrieg
country_event = {
	id = worldtension.1
	immediate = {log = "[GetDateText]: [Root.GetName]: event worldtension.1"}

	trigger = {
		GER = { has_war_with = FRA }
	}

	fire_only_once = yes

	hidden = yes

	option = {
		FRA = {
			add_named_threat = { threat = 50 name = second_weltkrieg_threat }
		}
	}
}
